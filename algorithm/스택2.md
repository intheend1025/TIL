def permutation(k):
    global visited
    
    if k == N:
        print(visited)
        
    for idx in range(k, N):
        visited[k], visited[idx] = visited[idx], visited[k]
        permutation(k+1)  # 처음 자리는 고정 이후 자리 바꿔보자
        visited[k], visited[idx] = visited[idx], visited[k]

### 부분집합

```python
arr = [1, 2, 3]
N = len(arr)
sel = [0] * N

def powerset(idx):
    if idx == N:
        for i in range(N):
            if sel[i]:
                print(arr[i], end=' ')
        print()
        return
            
    # 해당 원소를 뽑음   
    sel[idx] = 1
    powerset(idx+1)
    
    # 해당 원소 뽑지 않음
    sel[idx] = 0
    powerset(idx+1)
    
powerset(0)
```
